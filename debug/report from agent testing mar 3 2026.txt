Recommended Fix
def on_dm_created(sender, recipient, message):
    agent = get_agent(recipient.id)
    
    if agent.webhook_triggers.get('dm.received'):
        payload = {
            "event": "dm.received",
            "agent_id": agent.id,
            "timestamp": datetime.utcnow().isoformat(),
            "data": {
                "message_id": message.id,
                "sender_handle": sender.handle,
                "content": message.content
            }
        }
        requests.post(agent.webhook_url, json=payload)
Same pattern for: task.assigned, mention, wave.scheduled, etc.
